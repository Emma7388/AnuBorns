<script is:inline>
  document.addEventListener("DOMContentLoaded", () => {
    const modal = document.getElementById("loginModal");
    const openButtons = document.querySelectorAll(".nav-button, .primary");
    const closeBtn = document.querySelector(".close-btn");
    const overlay = document.querySelector(".modal-overlay");

    const tabs = document.querySelectorAll(".role-tabs button");
    const form = modal.querySelector("form");

    // abrir
    openButtons.forEach((btn) => {
      btn.addEventListener("click", () => {
        modal.classList.add("active");
      });
    });

    // cerrar
    const closeModal = () => modal.classList.remove("active");

    closeBtn?.addEventListener("click", closeModal);
    overlay?.addEventListener("click", closeModal);

    document.addEventListener("keydown", (e) => {
      if (e.key === "Escape") closeModal();
    });

    // tabs dinámicos
    const extraField = modal.querySelector(".extra-field");
    const submitBtn = modal.querySelector(".primary-btn");

    tabs.forEach((tab) => {
      tab.addEventListener("click", () => {
        tabs.forEach((t) => t.classList.remove("active"));
        tab.classList.add("active");

        if (tab.textContent.trim() === "Profesional") {
          submitBtn.textContent = "Ingresar como Profesional";
        } else {
          extraField.innerHTML = "";
          submitBtn.textContent = "Ingresar como Cliente";
        }
      });
    });
  });
</script>
